// 🧠 Emotional memory integration
async function recordInterviewEmotion(userId: string, sessionData: any) {
  try {
    // Record emotional state from interview
    await recordEmotionalState(userId, {
      emotion: sessionData.emotion || "neutral",
      intensity: sessionData.emotionalIntensity || 5,
      context: "Interview session",
      metadata: {
        sessionId: sessionData.sessionId,
        interviewType: sessionData.interviewType || "technical",
        score: sessionData.score || 0,
        timestamp: new Date().toISOString()
      }
    });
    
    console.log(`✅ Recorded emotion for ${userId}: ${sessionData.emotion || "neutral"}`);
    return true;
  } catch (error) {
    console.error("❌ Failed to record emotion:", error);
    return false;
  }
}
